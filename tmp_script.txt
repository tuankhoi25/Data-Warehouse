CREATE DATABASE db_in_ch;

CREATE TABLE db_in_ch.category
(
    id UInt64,
    category_name Nullable(String),
    created_at Nullable(Date),
    updated_at Nullable(Date)
)
ENGINE = PostgreSQL('postgres:5432', 'oltp', 'category', 'clickhouse_user', 'clickhouse_password');

CREATE TABLE db_in_ch.staging_category
(
    id UInt64,
    category_name Nullable(String),
    created_at Nullable(Date),
    updated_at Nullable(Date)
)
ENGINE = MergeTree
ORDER BY id;

INSERT INTO db_in_ch.staging_category
SELECT * FROM db_in_ch.category
WHERE id > 4;
